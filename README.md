def birthday_to_words(birthday_str):
    birthday = datetime.datetime.strptime(birthday_str, '%m-%d')
    month_name = birthday.strftime('%B')  # –ü–æ–ª—É—á–∞–µ–º –Ω–∞–∑–≤–∞–Ω–∏–µ –º–µ—Å—è—Ü–∞
    month_name = month_name.lower().capitalize()  # –ü—Ä–µ–æ–±—Ä–∞–∑—É–µ–º –ø–µ—Ä–≤—É—é –±—É–∫–≤—É –≤ –≤–µ—Ä—Ö–Ω–∏–π —Ä–µ–≥–∏—Å—Ç—Ä
    month_name = month_name.replace("January", "—è–Ω–≤–∞—Ä—è").replace("February", "—Ñ–µ–≤—Ä–∞–ª—è").replace("March", "–º–∞—Ä—Ç–∞") \
        .replace("April", "–∞–ø—Ä–µ–ª—è").replace("May", "–º–∞—è").replace("June", "–∏—é–Ω—è").replace("July", "–∏—é–ª—è") \
        .replace("August", "–∞–≤–≥—É—Å—Ç–∞").replace("September", "—Å–µ–Ω—Ç—è–±—Ä—è").replace("October", "–æ–∫—Ç—è–±—Ä—è") \
        .replace("November", "–Ω–æ—è–±—Ä—è").replace("December", "–¥–µ–∫–∞–±—Ä—è")  # –ü–µ—Ä–µ–≤–æ–¥–∏–º –Ω–∞ —Ä—É—Å—Å–∫–∏–π —è–∑—ã–∫
    return f'{birthday.day} {month_name}'
